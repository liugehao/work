SELECT
  id,
  cxbm,
  cxmc,
  sfd,
  mdd
FROM
  cxxx
WHERE mdd = 210000

SELECT
  id,
  cxbm,
  cxmc,
  sfd,
  mdd
FROM
  cxxx
WHERE mdd = '210000'
